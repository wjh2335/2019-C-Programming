※add
	int add(int a,int b)
	{
		return a+b;
	}
* * *
※sub
	int sub(int a,int b)
	{
		return a-b;
	}
* * *
※mul
	int mul(int a,int b)
	{
		return a*b;
	}
* * *
※div
	float div(int a,int b)
	{
		return (float)a/b;
	}
* * *
※mok
	int mok(int a,int b)
	{
		return a/b;
	}
* * *
※nam
	int nam(int a,int b)
	{
		return a%b;
	}
